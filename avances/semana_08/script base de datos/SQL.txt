-- TABLA CLIENTE
CREATE TABLE "CLIENTE" (
	"ID_CLIENTE"	INTEGER,
	"NOMBRE"	TEXT NOT NULL,
	"CORREO_ELECTRONICO"	TEXT NOT NULL,
	"TELEFONO"	TEXT,
	PRIMARY KEY("ID_CLIENTE" AUTOINCREMENT)
);

-- TABLA CONSULTA DETALLADA
CREATE TABLE "CONSULTA_DETALLADA" (
	"N_CONSULTA"	INTEGER,
	"MOTIVO_CONSULTA"	TEXT NOT NULL CHECK(LENGTH("MOTIVO_CONSULTA") <= 150),
	"EXAMEN_AUXILIAR"	TEXT NOT NULL CHECK(LENGTH("EXAMEN_AUXILIAR") <= 100),
	"TRATAMIENTO"	TEXT NOT NULL CHECK(LENGTH("TRATAMIENTO") <= 100),
	"DETALLES_EXTRAS"	TEXT NOT NULL CHECK(LENGTH("DETALLES_EXTRAS") <= 200),
	"FECHA"	DATE NOT NULL,
	"N_CHIP"	INTEGER NOT NULL,
	PRIMARY KEY("N_CONSULTA" AUTOINCREMENT),
	FOREIGN KEY("N_CHIP") REFERENCES "MASCOTA"("N_CHIP")
);

-- TABLA MASCOTA
CREATE TABLE "MASCOTA" (
	"N_CHIP"	INTEGER,
	"NOMBRE_MASCOTA"	TEXT NOT NULL,
	"ESPECIE"	TEXT NOT NULL,
	"RAZA"	TEXT,
	"PESO"	REAL,
	"FECHA_NACIMIENTO"	TEXT,
	"SEXO"	TEXT,
	"ID_CLIENTE"	INTEGER,
	PRIMARY KEY("N_CHIP"),
	FOREIGN KEY("ID_CLIENTE") REFERENCES "CLIENTE"("ID_CLIENTE")
);
